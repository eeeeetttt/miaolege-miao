## 1. 项目概述

### 1.1 项目简介

金猴火窝是一个专注于金融交易模拟挑战的平台，提供：
- MT4/MT5 模拟交易环境
- 跟单学习系统（星球跟单）
- 10关闯关挑战赛（K线征途）
- 多种赛事系统（天梯赛、大师邀请赛等）
- 社区交流功能（聊天大厅、论坛）
- EA产品商城

#### 4.2.1 用户表 (user_accounts)

```sql
CREATE TABLE user_accounts (
    -- 主键和认证
    user_id VARCHAR(36) PRIMARY KEY COMMENT '用户UUID，主键',
    email VARCHAR(255) NOT NULL UNIQUE COMMENT '邮箱，唯一',
    password_hash VARCHAR(255) NOT NULL COMMENT 'bcrypt加密的密码',
    
    -- 用户信息
    name VARCHAR(100) DEFAULT '用户' COMMENT '昵称',
    role ENUM('admin', 'user', 'vip') DEFAULT 'user' COMMENT '角色',
    coin_balance DECIMAL(15,2) DEFAULT 10000.00 COMMENT '星球币余额',
    avatar_url VARCHAR(500) DEFAULT NULL COMMENT '头像URL',
    email_verified TIMESTAMP NULL DEFAULT NULL COMMENT '邮箱验证时间',
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    
    -- 索引
    INDEX idx_email (email),
    INDEX idx_role (role),
    INDEX idx_created (created_at)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci COMMENT='用户账户表';
```

**TypeScript 类型定义：**

```typescript
export const userAccounts = mysqlTable('user_accounts', {
```

#### 7.1.1 NextAuth Handler

路径：`src/app/api/auth/[...nextauth]/route.ts`

```typescript
import NextAuth, { NextAuthOptions } from 'next-auth';
import CredentialsProvider from 'next-auth/providers/credentials';
import { compare } from 'bcryptjs';
import { db } from '@/lib/db';
import { userAccounts } from '@/lib/schema';
import { eq } from 'drizzle-orm';

export const authOptions: NextAuthOptions = {
  providers: [
    CredentialsProvider({
      name: 'credentials',
      credentials: {
        email: { label: '邮箱', type: 'email' },
        password: { label: '密码', type: 'password' },
      },
      async authorize(credentials) {
        if (!credentials?.email || !credentials?.password) {
          return null;
        }

        const user = await db.query.userAccounts.findFirst({
          where: eq(userAccounts.email, credentials.email),
        });

        if (!user) {
          return null;
        }

        const isPasswordValid = await compare(credentials.password, user.passwordHash);

        if (!isPasswordValid) {
          return null;
        }

        return {
          id: user.userId,
          email: user.email,
          name: user.name,
          role: user.role,
        };
      },
    }),
  ],
  callbacks: {
    async jwt({ token, user }) {
      if (user) {
        token.id = user.id;
        token.role = (user as any).role;
      }
      return token;
    },
    async session({ session, token }) {
      if (session.user) {
        (session.user as any).id = token.id;
        (session.user as any).role = token.role;
      }
      return session;
    },
  },
  pages: {
    signIn: '/login',
    error: '/login',
  },
  session: {
    strategy: 'jwt',
    maxAge: 30 * 24 * 60 * 60, // 30天
  },
  secret: process.env.NEXTAUTH_SECRET,
};

const handler = NextAuth(authOptions);
export { handler as GET, handler as POST };
```

```typescript
import { hash } from 'bcryptjs';
import { db } from '@/lib/db';
import { userAccounts } from '@/lib/schema';
import { eq } from 'drizzle-orm';

export async function POST(request: NextRequest) {
  try {
    const { email, password, name } = await request.json();

    // 验证必填字段
    if (!email || !password) {
      return NextResponse.json(
        { error: '邮箱和密码不能为空' },
        { status: 400 }
      );
    }

    // 检查邮箱是否已存在

      where: eq(userAccounts.email, email),
    });

    if (existingUser) {
      return NextResponse.json(
        { error: '该邮箱已注册' },
        { status: 400 }
      );
    }

    // 加密密码
    const passwordHash = await hash(password, 12);

    // 创建用户
    const userId = crypto.randomUUID();

      userId,
      email,
      passwordHash,
      name: name || '用户',
      role: 'user',
      coinBalance: defaultCoins,
      emailVerified: new Date(),
    });

    return NextResponse.json({
      success: true,
      message: '注册成功',
      userId,
    });
  } catch (error) {
    console.error('注册失败:', error);
    return NextResponse.json(
      { error: '注册失败，请稍后重试' },
      { status: 500 }
    );
  }
```

### 7.4 挑战 API

#### 7.4.1 挑战报名

路径：`src/app/api/challenge/register/route.ts`

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { getServerSession } from 'next-auth';
import { authOptions } from '@/lib/auth';
import { db } from '@/lib/db';
import { challengeRegistrations, userAccounts } from '@/lib/schema';
import { eq, and } from 'drizzle-orm';

export async function POST(request: NextRequest) {
  try {
    const session = await getServerSession(authOptions);

    if (!session?.user?.email) {
      return NextResponse.json({ error: '请先登录' }, { status: 401 });
    }

    const userId = (session.user as any).id;
    const { mode = 'paid' } = await request.json();

    // 检查是否已有进行中的挑战
    const existingChallenge = await db.query.challengeRegistrations.findFirst({
      where: and(
        eq(challengeRegistrations.userId, userId),
        eq(challengeRegistrations.status, 'active')
      ),
    });

    if (existingChallenge) {
      return NextResponse.json(
        { error: '您已有进行中的挑战' },
        { status: 400 }
      );
    }

    // 检查是否有待审核的申请
    const pendingChallenge = await db.query.challengeRegistrations.findFirst({
      where: and(
        eq(challengeRegistrations.userId, userId),
        eq(challengeRegistrations.status, 'pending')
      ),
    });

    if (pendingChallenge) {
      return NextResponse.json(
        { error: '您有待审核的申请，请等待审核' },
        { status: 400 }
      );
    }

    // 付费模式扣除星球币
    if (mode === 'paid') {
      const challengeFee = 1000; // 报名费

      const user = await db.query.userAccounts.findFirst({
        where: eq(userAccounts.userId, userId),
      });

      if (!user || Number(user.coinBalance) < challengeFee) {
        return NextResponse.json(
          { error: '星球币不足，需要 1000 星球币' },
          { status: 400 }
        );
      }

      // 扣除星球币
      await db
        .update(userAccounts)
        .set({
          coinBalance: Number(user.coinBalance) - challengeFee,
        })
        .where(eq(userAccounts.userId, userId));
    }

    // 创建报名申请
    await db.insert(challengeRegistrations).values({
      userId,
      status: 'pending',
      currentLevel: 1,
    });

    return NextResponse.json({
      success: true,
      message: '报名成功，请等待管理员审核',
    });
  } catch (error) {
    console.error('挑战报名失败:', error);
    return NextResponse.json({ error: '报名失败' }, { status: 500 });
  }
}
```
